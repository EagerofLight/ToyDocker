# ToyDocker
Toy-level docker, learning from web, for practice
# Implemented functions
1. ./toy-docker init
2. ./toy-docker ps
3. ./toy-docker logs
4. ./toy-docker commit
5. ./toy-docker run
   1. enable tyy: -ti
   2. volume: -v
   3. memory limit: -m
   4. cpushare limit: -cpushare
   5. cpuset limit: -cpuset
   6. container name: -name
